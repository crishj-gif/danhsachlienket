Danh sách liên kết

Học phần: Kỹ thuật lập trình

1. Lý thuyết trọng tâm

Danh sách liên kết (Linked List) là một cấu trúc dữ liệu động, trong đó các phần tử được lưu trữ tại các vị trí rời rạc trong bộ nhớ và được liên kết với nhau bằng các con trỏ.

Đặc điểm Mảng (Array) DSLK (Linked List)
Kích thước Cố định Linh hoạt
Truy cập Ngẫu nhiên O(1) Tuần tự O(n)
Thao tác Thêm/Xóa Chậm (Dời phần tử) Nhanh (Đổi con trỏ)

2. Cấu trúc & Khởi tạo

Mỗi phần tử trong danh sách được gọi là một Node, bao gồm phần dữ liệu và phần liên kết.

#include <stdio.h>
#include <stdlib.h>

// Dinh nghia cau truc Node
struct Node {
    int data;
    struct Node *next;
};
typedef struct Node NODE;

// Dinh nghia cau truc quan ly danh sach
struct List {
    NODE *pHead;
    NODE *pTail;
};
typedef struct List LIST;

// Khoi tao danh sach rong
void Init(LIST *l) {
    l->pHead = l->pTail = NULL;
}

// Tao mot Node moi
NODE* CreateNode(int x) {
    NODE *p = (NODE*)malloc(sizeof(NODE));
    if (p == NULL) return NULL;
    p->data = x;
    p->next = NULL;
    return p;
}

3. Thao tác thêm vào danh sách

Thêm vào đầu (AddHead)

void AddHead(LIST *l, NODE *p) {
    if (l->pHead == NULL) {
        l->pHead = l->pTail = p;
    } else {
        p->next = l->pHead;
        l->pHead = p;
    }
}

Thêm vào cuối (AddTail)

void AddTail(LIST *l, NODE *p) {
    if (l->pHead == NULL) {
        l->pHead = l->pTail = p;
    } else {
        l->pTail->next = p;
        l->pTail = p;
    }
}

4. Thao tác Xóa

Hay ra thi

Xóa đầu danh sách

void RemoveHead(LIST *l) {
    if (l->pHead == NULL) return;
    NODE *p = l->pHead;
    l->pHead = l->pHead->next;
    free(p);
    if (l->pHead == NULL) l->pTail = NULL;
}

Xóa Node có giá trị X

void RemoveX(LIST *l, int x) {
    if (l->pHead == NULL) return;
    if (l->pHead->data == x) {
        RemoveHead(l);
        return;
    }
    NODE *prev = l->pHead;
    NODE *curr = l->pHead->next;
    while (curr != NULL) {
        if (curr->data == x) {
            prev->next = curr->next;
            if (curr == l->pTail) l->pTail = prev;
            free(curr);
            return;
        }
        prev = curr;
        curr = curr->next;
    }
}

5. Thống kê & Tìm kiếm

Tìm giá trị Max trong DSLK

int FindMax(LIST l) {
    if (l.pHead == NULL) return -1;
    int max = l.pHead->data;
    for (NODE *p = l.pHead->next; p != NULL; p = p->next) {
        if (p->data > max) max = p->data;
    }
    return max;
}

Tính trung bình cộng các số lẻ

float AverageOdd(LIST l) {
    int sum = 0, count = 0;
    for (NODE *p = l.pHead; p != NULL; p = p->next) {
        if (p->data % 2 != 0) {
            sum += p->data;
            count++;
        }
    }
    if (count == 0) return 0;
    return (float)sum / count;
}

6. Bài toán Quản lý Sinh Viên

Dạng bài thực tế

Cấu trúc này thường dùng cho câu 2 trong đề thi (quản lý một đối tượng thực tế).

struct SinhVien {
    char maSV[15];
    char tenSV[50];
    float diem;
};
typedef struct SinhVien SV;

struct NodeSV {
    SV data;
    struct NodeSV *pNext;
};
typedef struct NodeSV NODESV;

// Tim sinh vien co diem cao nhat
void PrintBestStudent(NODESV *pHead) {
    if (pHead == NULL) return;
    float maxDiem = pHead->data.diem;
    for (NODESV *p = pHead; p != NULL; p = p->pNext) {
        if (p->data.diem > maxDiem) maxDiem = p->data.diem;
    }
    printf("Cac sinh vien diem cao nhat:\n");
    for (NODESV *p = pHead; p != NULL; p = p->pNext) {
        if (p->data.diem == maxDiem) {
            printf("%s - %s\n", p->data.maSV, p->data.tenSV);
        }
    }
}

7. Thao tác nâng cao

Sắp xếp tăng dần (GIỮ NGUYÊN)

void Sort(LIST *l) {
    for (NODE *p = l->pHead; p != NULL; p = p->next) {
        for (NODE *q = p->next; q != NULL; q = q->next) {
            if (p->data > q->data) {
                int tmp = p->data;
                p->data = q->data;
                q->data = tmp;
            }
        }
    }
}

Đảo ngược danh sách (Kỹ thuật 3 con trỏ)

void Reverse(LIST *l) {
    NODE *prev = NULL, *curr = l->pHead, *next = NULL;
    l->pTail = l->pHead;
    while (curr != NULL) {
        next = curr->next;
        curr->next = prev;
        prev = curr;
        curr = next;
    }
    l->pHead = prev;
}

8. Các lỗi sinh viên hay mắc

  • Quên giải phóng bộ nhớ: Luôn sử dụng free() khi xóa Node để tránh rò rỉ RAM.
  • Mất dấu pHead: Sử dụng trực tiếp l->pHead để duyệt danh sách sẽ làm mất địa chỉ đầu. Hãy dùng biến tạm.
  • Truy cập con trỏ NULL: Luôn kiểm tra p != NULL trước khi gọi p->next.

9. Đề thi mẫu (Tham khảo)

Câu 1: Viết hàm tính tổng các số chẵn trong Danh sách liên kết đơn các số nguyên.

int SumEven(LIST l) {
    int sum = 0;
    for (NODE *p = l.pHead; p != NULL; p = p->next) {
        if (p->data % 2 == 0) {
            sum += p->data;
        }
    }
    return sum;
}

Câu 2: Viết hàm đếm xem có bao nhiêu số nguyên tố trong DSLK.

int isPrime(int n) {
    if (n < 2) return 0;
    for (int i = 2; i * i <= n; i++)
        if (n % i == 0) return 0;
    return 1;
}

int CountPrimes(LIST l) {
    int count = 0;
    for (NODE *p = l.pHead; p != NULL; p = p->next) {
        if (isPrime(p->data)) count++;
    }
    return count;
}
